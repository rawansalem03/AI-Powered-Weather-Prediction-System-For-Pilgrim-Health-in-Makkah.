function updateDate() {
    const now = new Date();
    const days = ['الأحد', 'الإثنين', 'الثلاثاء', 'الأربعاء', 'الخميس', 'الجمعة', 'السبت'];
    document.getElementById('dayName').innerText = days[now.getDay()];
    document.getElementById('gregorianDate').innerText = now.toLocaleDateString('ar-SA', {year:'numeric', month:'long', day:'numeric'}) + ' م';

}

// 2. دالة جلب البيانات الحقيقية من قاعدة البيانات
async function refreshDashboardData() {
    try {
        // أ. جلب إجمالي التنبؤات (عدد الحجاج الذين استخدموا النظام)
        const { count: totalPredictions } = await _supabase
            .table('predictions')
            .select('*', { count: 'exact', head: true });

        // ب. جلب عدد بلاغات الطوارئ النشطة
        const { count: activeEmergencies } = await _supabase
            .table('emergency_team')
            .select('*', { count: 'exact', head: true })
            .eq('status', 'Active');

        // ج. جلب حالة الموارد (الأسرة والحرارة) من آخر تحديث
        const { data: resources } = await _supabase
            .table('health_resources')
            .select('*')
            .limit(1);

        if (resources && resources[0]) {
            const res = resources[0];
            const occupancyPerc = Math.round((res.occupied_beds / res.total_beds) * 100);

            // تحديث القيم في الواجهة
            document.getElementById('tempVal').innerText = "38°م"; // يمكن جلبها من API الطقس
            document.getElementById('humVal').innerText = "45%";

            // تحديث شريط الخطر بناءً على نسبة إشغال الأسرة أو التنبؤات
            const riskLevel = occupancyPerc > 70 ? 'عالية' : (occupancyPerc > 40 ? 'متوسطة' : 'منخفضة');
            document.getElementById('riskVal').innerText = riskLevel;
            document.getElementById('riskVal').className = 'metric-value ' + (occupancyPerc > 70 ? 'sev-high' : 'sev-low');

            // تحريك مؤشر شريط الخطر
            document.getElementById('riskThumb').style.left = occupancyPerc + '%';
        }

        // د. تحديث عدد الحالات في واجهة المسؤول
        console.log(`إحصائيات: حجاج: ${totalPredictions}, بلاغات: ${activeEmergencies}`);

    } catch (err) {
        console.error("خطأ في جلب بيانات لوحة التحكم:", err);
    }
}

// 3. دالة إرسال تنبيه عام (لجدول notifications)
async function sendSystemNotification(msg) {
    const { error } = await _supabase.table('notification').insert([
        { message: msg, timestamp: new Date() }
    ]);
    if(!error) alert("تم إرسال التنبيه لجميع الحجاج والجمهور");
}
